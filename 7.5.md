### 1、1、1函数重载

作用：可以让函数名字重复

```cpp
#include <iostream>
using namespace std;

void print(int x) {
    cout << "int: " << x << endl;
}

void print(double x) {
    cout << "double: " << x << endl;
}

void print(string s) {
    cout << "string: " << s << endl;
}

int main() {
    print(10);        // 调用 print(int)
    print(3.14);      // 调用 print(double)
    print("hello");   // 调用 print(string)

    return 0;
}
```

函数的参数、类型、顺序不同，但是在同一作用域

# 2、析构和构造

构造函数：调用的时候完成对象初始化；析构函数：销毁对象

## 2、1构造函数

初始化成员属性

```cpp
#include<iostream>
#include<string>
using namespace std;
class Cake{
public:
    string name;
    int price;
    Cake(string n, int p){//从堆区申请空间
        name = n;
        price = p;
    }//构造函数与类名相同，与python相似
};
int main(){
    Cake cake1("Chocolate", 20);
    Cake cake2("Vanilla", 15);
    cout << "Cake 1: " << cake1.name << ", Price: $" << cake1.price << endl;
    cout << "Cake 2: " << cake2.name << ", Price: $" << cake2.price << endl;
    return 0;
}
```

class Up(nn.Module):
    def __init__(self, in_channel, out_channel):
        super().__init__()
        self.conv1=nn.Sequential(
            nn.ConvTranspose2d(in_channel, in_channel//2, kernel_size=2, stride=2),
        )
        self.conv=DoubleConv(in_channel, out_channel)
    def forward(self,x1,x2):
        x1=self.conv1(x1)
        out=torch.cat([x2,x1],dim=1)
        return self.conv(out)
class Down(nn.Module):
     def __init__(self, in_channel, out_channel):
         super().__init__()
         self.conv1=nn.Sequential(
             nn.MaxPool2d(2),
             DoubleConv(in_channel, out_channel)
         )
     def forward(self,x):
         return self.conv1(x)
class DoubleConv(nn.Module):
    def __init__(self, in_channel, out_channel):
        super().__init__()
        self.conv1=nn.Sequential(
            nn.Conv2d(in_channel, out_channel, kernel_size=3, stride=1, padding=1),
            nn.BatchNorm2d(out_channel),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channel, out_channel, kernel_size=3, stride=1, padding=1),
            nn.BatchNorm2d(out_channel),
            nn.ReLU(inplace=True)
        )
    def forward(self,x):
        return self.conv1(x)

class UNet(nn.Module):
    def __init__(self,n_channels,n_classes):
        super().__init__()
        self.n_channels = n_channels
        self.n_classes = n_classes
        self.inc = DoubleConv(n_channels, 4)
        self.down1 = Down(4, 8)
        self.down2 = Down(8, 16)
        self.down3 = Down(16, 32)
        self.down4 = Down(32, 64)
        self.up1 = Up(64, 32)
        self.up2 = Up(32, 16)
        self.up3 = Up(16, 8)
        self.up4 = Up(8, 4)
        self.outc = nn.Conv2d(4, n_classes, kernel_size=1)
    def forward(self,x):
        x1=self.inc(x)
        x2=self.down1(x1)
        x3=self.down2(x2)
        x4=self.down3(x3)
        x5=self.down4(x4)
        x=self.up1(x5,x4)
        x=self.up2(x,x3)
        x=self.up3(x,x2)
        x=self.up4(x,x1)
        logits=self.outc(x)
        return logits
